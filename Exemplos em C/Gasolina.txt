#include <stdio.h>
#include <stdlib.h>

int main()
{
    float vm, gasolina, distancia, tempo;

    printf("Digite a velocidade média do motorista em Km \n");
    scanf ("%f",&vm);
    printf ("Digite o tempo em horas \n");
    scanf ("%f",&tempo);

    distancia = vm*tempo;
    printf ("A distancia percorrida foi %f\n", distancia);
    gasolina = distancia/8;
    if (0<=gasolina && gasolina<=20) {
        printf ("O tanque esta cheio");
    }

    else{

        if (21<=gasolina && gasolina<=40) printf ("O tanque esta comprometido");
        else printf ("Reabasteca imediatamente!");


    }
    return 0;
}
