#include <stdio.h>
#include <stdlib.h>

int main()
{
    float x,c,f;
    char t;

    printf("Digite a temperatura \n");
    scanf("%f", &x);
    fflush(stdin);
    printf("Digite a conversao [Celsius 'c'/ Farenheit 'f']");
    scanf("%c", &t);


    if (t == 'c'){
        f= (x*1.8)+32;
        printf("A temperatura em Farenheit e: %f", f);
    }else{
        if (t == 'f'){
            c= (x-32)/1.8;
            printf("A temperatura em Celsius e: %f", c);
        }else{
            printf("Erro");
        }
    }
    return 0;
}
